27 /// @brief SOCKS-specific error codes
28 enum class error
29 {
30  /// @brief No error.
31  none = 0,
32 
33  /// @brief SOCKS version mismatch between server and client.
34  ///
35  /// Used in case the client receives a packet with a different
36  // server major version.
37  invalid_version,
38 
39  /// @brief Server supports none of our authentication methods.
40  ///
41  /// Used in case the server rejects all of our proposed authentication
42  /// methods
43  ///
44  /// @see async_greet
45  no_acceptable_auth_method,
46 
47  /// @brief Authentication type version mismatch between client and server.
48  ///
49  /// The version of the agreed-on authentication scheme is different on
50  /// the server.
51  invalid_auth_version,
52 
53  /// @brief The server rejected our login attempt.
54  login_failed,
55 
56  /// @brief The SOCKS @c command we sent was rejected.
57  ///
58  /// The server doesn't understand the @c command we sent.
59  command_not_supported,
60 };
