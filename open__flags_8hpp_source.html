28 /// @brief Specifies semantics for opening files.
29 ///
30 /// This enum of bitmask values controls the behaviour of the @ref open.
31 /// @c open_flags meets the requirements
32 /// of [BitmaskType](http://en.cppreference.com/w/cpp/concept/BitmaskType).
33 ///
34 /// open() converts these to their platform's native equivalent (if possible).
35 /// Flags that are only available on certain platforms are marked as such.
36 ///
37 /// There are 4 categories of flags:
38 /// * File access flags (@c access_read, ...)
39 /// * File creation disposition flags (@c create_new, ...)
40 /// * Sharing mode flags (@c exclusive_read, ...)
41 /// * Special flags (@c async, ...)
42 ///
43 /// File creation disposition flags are mutually exclusive.
44 /// Specifying more than one is an error.
45 enum class open_flags
46 {
47  /// No options are set.
48  none = 0,
49 
50  // access-mode flags
51 
52  /// Request read access to the file.
53  access_read = 1 << 0,
54 
55  /// Request write access to the file.
56  access_write = 1 << 1,
57 
58  /// Request read and write access to the file.
59  ///
60  /// Note that this is equivalent to:
61  /// <tt>access_read | access_write</tt>
62  access_read_write = access_read | access_write,
63 
64  // creation-mode flags
65 
66  /// Attempt to create a new file. Fail if the file already exists.
67  create_new = 1 << 2,
68 
69  /// Always create a new file. If the file already exists, it is truncated.
70  create_always = 1 << 3,
71 
72  /// Attempt to open an existing file. Fail if no such file exists.
73  open_existing = 1 << 4,
74 
75  /// Always open the file. If no such file exists, it is created.
76  open_always = 1 << 5,
77 
78  /// Attempt to open and truncate the file. Fail if no such file exists.
79  /// This requires the access_write bit to be set.
80  truncate_existing = 1 << 6,
81 
82  // sharing
83 
84  /// Request exclusive read access to the file.
85  ///
86  /// @note Currently only implemented on Windows.
87  exclusive_read = 1 << 7,
88 
89  /// Request exclusive write access to the file.
90  ///
91  /// @note Currently only implemented on Windows.
92  exclusive_write = 1 << 8,
93 
94  // special
95 
96 #if !defined(ASIOEXT_IS_DOCUMENTATION)
97  // XXX: Intentionally undocumented, Windows only.
98  /// Open the file for asynchronous operations.
99  ///
100  /// @warning A handle created with this flag cannot be used in combination
101  /// with @c file_handle's I/O functions.
102  internal_async = 1 << 9,
103 #endif
104 };
105
