54 /// @brief SOCKS login authentication methods.
55 ///
56 /// This enum contains the authentication methods
57 /// listed in the protocol specification.
58 enum class auth_method : uint8_t
59 {
60  /// @brief No authentication.
61  ///
62  /// If sent by the client, indicates that the client
63  /// supports/allows anonymous access to a SOCKS server.
64  none = 0,
65 
66  /// @brief GSSAPI authentication.
67  ///
68  /// @note There is currently no public function that handles the
69  /// login challenge for this authentication type.
70  gssapi = 1,
71 
72  /// @brief Username + Password authentication.
73  ///
74  /// Indicates that the client wishes to authenticate the user
75  /// with a username and password pair.
76  ///
77  /// @see async_login
78  username_password = 2,
79 
80  /// @brief Server response if no acceptable method was given.
81  ///
82  /// This special enum value is used in case the server rejects all
83  /// authentication methods it was given.
84  no_acceptable = 0xff
85 };
