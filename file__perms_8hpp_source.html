26 /// @brief Names for permissions.
27 ///
28 /// This type represents file access permissions.
29 /// @c file_perms meets the requirements
30 /// of [BitmaskType](http://en.cppreference.com/w/cpp/concept/BitmaskType).
31 enum class file_perms
32 {
33  /// No permissions are set.
34  none = 0,
35 
36  /// S_IRUSR, Read permission, owner
37  owner_read = 0400,
38 
39  /// S_IWUSR, Write permission, owner
40  owner_write = 0200,
41 
42  /// S_IXUSR, Execute/search permission, owner
43  owner_exec = 0100,
44 
45  /// S_IRWXU, Read, write, execute/search by owner
46  owner_all = 0700,
47 
48  /// S_IRGRP, Read permission, group
49  group_read = 040,
50 
51  /// S_IWGRP, Write permission, group
52  group_write = 020,
53 
54  /// S_IXGRP, Execute/search permission, group
55  group_exec = 010,
56 
57  /// S_IRWXG, Read, write, execute/search by group
58  group_all = 070,
59 
60  /// S_IROTH, Read permission, others
61  others_read = 04,
62 
63  /// S_IWOTH, Write permission, others
64  others_write = 02,
65 
66  /// S_IXOTH, Execute/search permission, others
67  others_exec = 01,
68 
69  /// S_IRWXO, Read, write, execute/search by others
70  others_all = 07,
71 
72  /// All users have read, write, and execute/search permissions.
73  ///
74  /// Equivalent to `owner_all | group_all | others_all`
75  all = 0777,
76 
77  /// S_ISUID, Set-user-ID on execution
78  set_uid = 04000,
79 
80  /// S_ISGID, Set-group-ID on execution
81  set_gid = 02000,
82 
83  /// S_ISVTX,
84  /// * **POSIX XSI**: On directories, restricted deletion flag
85  /// * **V7**: 'sticky bit': save swapped text even after use
86  /// * **SunOS**: On non-directories: don't cache this file
87  /// * **SVID-v4.2**: On directories: restricted deletion flag
88  ///
89  /// @see http://en.wikipedia.org/wiki/Sticky_bit
90  sticky_bit = 01000,
91 
92  /// Default permissions for newly created files (like @c DEFFILEMODE).
93  ///
94  /// Set to @c 0666 (
95  /// `owner_read | owner_write | group_read | group_write |
96  /// others_read | others_write`) on all systems.
97  ///
98  /// @note Actual file permissions might differ, as these permissions
99  /// are modified by the process' umask.
100  create_default = 0666,
101 
102  /// All valid permission bits
103  ///
104  /// Equivalent to `all | set_uid | set_gid | sticky_bit`.
105  mask = 07777,
106 
107  /// Unknown permissions
108  unknown = 0xFFFF,
109 };

113 /// @brief Options to control the behavior of
114 /// @ref asioext::file_handle::permissions(file_perms,file_perm_options)
115 ///
116 /// This enum represents options for the file permissions modifier.
117 /// @c file_attr_options meets the requirements
118 /// of [BitmaskType](http://en.cppreference.com/w/cpp/concept/BitmaskType).
119 enum class file_perm_options
120 {
121  none = 0,
122 
123  /// Replace existing permissions.
124  replace = 1 << 0,
125 
126  /// Replace permissions with the bitwise OR
127  /// of the argument and the current permissions.
128  add = 1 << 1,
129 
130  /// Replace permissions with the bitwise AND
131  /// of the negated argument and the current permissions.
132  remove = 1 << 2,
133 };
