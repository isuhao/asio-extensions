8 /// @defgroup buildopts Build Options
9 /// Defining one of the following macros changes AsioExt behaviour.
10 /// By default, none of them are defined.
11 ///
12 /// @{
13 
14 /// @brief Use Boost.Asio instead of standalone Asio.
15 ///
16 /// If defined, AsioExt will use Boost.Asio headers and types.
17 #define ASIOEXT_USE_BOOST_ASIO
18 
19 /// @brief Build/Use AsioExt in a seperate compilation unit.
20 ///
21 /// This flag **must** be defined when building and consuming the library.
22 ///
23 /// Additionally, one compilation unit must
24 /// <code>\#include &lt;asioext/impl/src.cpp&gt;</code>.
25 #define ASIOEXT_SEPARATE_COMPILATION
26 
27 /// @brief Build/Use AsioExt as a shared library.
28 ///
29 /// This flag **must** be defined when building and consuming the library.
30 ///
31 /// Additionally, the shared library must have one CU that contains the
32 /// following line:
33 /// <code>\#include &lt;asioext/impl/src.cpp&gt;</code>.
34 ///
35 /// @note On Windows, AsioExt will declare functions with
36 /// <code>__declspec(dllimport/dllexport)</code>.
37 #define ASIOEXT_DYN_LINK
38 
39 /// @brief On Windows, assume <code>const char*</code> filenames
40 /// are UTF-8 encoded.
41 ///
42 /// This macro defines whether <tt>const char*</tt> filenames are encoded
43 /// in the system ANSI code page (if unset) or UTF-8 (if set).
44 #define ASIOEXT_WINDOWS_USE_UTF8_FILENAMES
45 
46 /// @brief Disable the use of Boost.Filesystem
47 ///
48 /// This macro disables the use of Boost.Filesystem, even if Boost is used
49 /// and the used version has a compatible Boost.Filesystem version.
50 #define ASIOEXT_DISABLE_BOOST_FILESYSTEM
51 
52 /// @brief Disable C++11 rvalue reference support.
53 ///
54 /// This macro disables the use of rvalue references, regardless of
55 /// compiler support.
56 #define ASIOEXT_DISABLE_MOVE
57 
58 /// @brief Disable C++11 variadic template support.
59 ///
60 /// This macro disables the use of variadic templates, regardless of
61 /// compiler support.
62 #define ASIOEXT_DISABLE_VARIADIC_TEMPLATES
63 
64 /// @brief Disable C++11 @c noexcept support.
65 ///
66 /// This macro disables the use of @c noexcept, regardless of
67 /// compiler support.
68 #define ASIOEXT_DISABLE_NOEXCEPT
69 
70 /// @brief Disable file flags support.
71 ///
72 /// This macro disables file flags, regardless of platform support.
73 /// Attributes passed to @c open() are silently ignored, while attempts
74 /// to query or modify file attributes fail.
75 #define ASIOEXT_DISABLE_FILE_FLAGS
76 
77 /// @brief Disable <code>\#pragma once</code> support.
78 ///
79 /// This macro disables the use of <code>\#pragma once</code>, regardless of
80 /// compiler support.
81 #define ASIOEXT_DISABLE_PRAGMA_ONCE
82 
83 /// @}
